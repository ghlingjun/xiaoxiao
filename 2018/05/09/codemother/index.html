    How do I get set up?

    • еҲӣе»әжЁЎжқҝж–Үд»¶еӨ№

      е°Ҷжң¬йЎ№зӣ®дёӯcodemotherзӣ®еҪ•еӨҚеҲ¶еҲ°иҮӘе·ұзҡ„йЎ№зӣ®зҡ„ж №зӣ®еҪ•дёӢгҖӮ

      templateдёӢзҡ„жЁЎжқҝеҸҜж №жҚ®е®һйҷ…йңҖиҰҒиҝӣиЎҢдҝ®ж”№гҖҒж–°еўһе’ҢеҲ йҷӨпјҢжЁЎзүҲиҜӯиЁҖжҳҜ freemarkerпјҢжһҒжҳ“дёҠжүӢгҖӮ

    • еҲӣе»ә schema зӣ®еҪ•

      schemaзӣ®еҪ•з”ЁдәҺеӯҳж”ҫдёҡеҠЎиЎЁеҜ№еә”зҡ„xmlй…ҚзҪ®ж–Үд»¶пјҲз”ҹжҲҗж–№жі•йҷ„еңЁж–ҮжЎЈжң«пјүпјҢcodemotherеҹәдәҺжӯӨй…ҚзҪ®ж–Үд»¶д»ҘеҸҠжЁЎжқҝз”ҹжҲҗд»Јз ҒгҖӮ

      еңЁйЎ№зӣ®жҲ–жЁЎеқ—дёӯзҡ„resourcesзӣ®еҪ•дёӢеҲӣе»әж–Үд»¶еӨ№com/ethanxx/codemotherпјҢе…¶дёӢеҢ…еҗ« dbschemaж–Үд»¶еӨ№д»ҘеҸҠglobal.propertiesж–Үд»¶пјҢ

      dbschemaж–Үд»¶еӨ№з”ЁдәҺеӯҳж”ҫдёҡеҠЎиЎЁеҜ№еә”зҡ„xmlй…ҚзҪ®ж–Үд»¶,global.propertiesй…ҚзҪ®дҝЎжҒҜе®һдҫӢеҰӮдёӢпјҡ

      # й»ҳи®ӨеҚіеҸҜпјҢж— йңҖжӣҙж”№

outputPagePath=src/main/webapp/WEB-INF/page/

resourceOutputPath=src/main/resources

    • й…ҚзҪ®mavenдҫқиө–

      еңЁйЎ№зӣ®жҲ–иҖ…жЁЎеқ—pom.xmlдёӯй…ҚзҪ®жҸ’д»¶дҫқиө–пјҡ

      <!--codemother жҸ’д»¶-->
<build>
    <plugins>
        <plugin>
            <groupId>com.ethanxx</groupId>
            <artifactId>codemother-maven-plugin</artifactId>

        </plugin>
    </plugins>
</build>

      еҖҳиӢҘдҪ зҡ„йЎ№зӣ®жҳҜеҚ•жЁЎеқ—пјҢжү§иЎҢе‘Ҫд»Өпјҡmvn codemother:code-generate -DtemplatePath=./codemother/template, иҝҷжҳҜеӣ дёәtemplatePathй»ҳи®ӨеҖјдёә../codemother/template, з”ЁдәҺеӨҡжЁЎеқ—йЎ№зӣ®гҖӮ

    Schemaж–Үд»¶з”ҹжҲҗж–№жі•

    • йҖҡиҝҮSchemaGeneratorе·Ҙе…·зұ»з”ҹжҲҗ

      е·Ҙе…·зұ»йҖҡиҝҮи§Јжһҗй…ҚзҪ®ж–Үд»¶codemother.propertiesпјҢиҝһжҺҘж•°жҚ®еә“пјҢж №жҚ®й…ҚзҪ®дёӯеҲ¶е®ҡзҡ„ж•°жҚ®иЎЁеҗҚпјҢз”ҹжҲҗеҜ№еә”зҡ„schemaж–Үд»¶

    • й…ҚзҪ®schemaж–Үд»¶

      з”ҹжҲҗзҡ„schemaж–Үд»¶еҸҜд»ҘзӣҙжҺҘдҪҝз”ЁпјҢдҪҶеҸҜд»Ҙж №жҚ®йңҖиҰҒеҒҡй«ҳзә§й…ҚзҪ®пјҢе…·дҪ“й…ҚзҪ®еҸӮж•°иҜҙжҳҺеҰӮдёӢпјҡ

          <schema tableName="ACHIEVEMENT_ALLOCATION" modelName="AchievementAllocation" paged="true" noController="true">
        <key name="id" type="integer" length="10" generator="native"></key>
    
        <column name="account_number" type="varchar" length="64"></column>
    
        <default-condition>
            <match column="user_id" param="userId"/>
            <match column="status_cd" param="statusCd"/>
            <like column="username"/>
            <match-value column="record_flag" value="1"/>
            <between column="last_login_time" javaType="java.sql.Timestamp" low="low" high="high"/>
            <in column="id"/>
        </default-condition>
    
        <default-order>
            <order-by column="username" asc="false"/>
        </default-order>
    
        <alias name="user"/>
    </schema>

      • ж Үзӯҫ key жҸҸиҝ°дәҶиЎЁзҡ„дё»й”®пјҢеҢ…еҗ«еұһжҖ§ name,type,length зӯү; еҪ“дё»й”®дёәеәҸеҲ—ж—¶пјҢж·»еҠ  generator еұһжҖ§пјҢеҖјдёә native;

      • ж Үзӯҫ match иЎЁзӨәжҹҘиҜўж—¶зІҫзЎ®еҢ№й…Қпјӣ

      • ж Үзӯҫ not-match иЎЁзӨәдёҚеҢ№й…ҚжҹҘиҜўпјӣ

      • ж Үзӯҫ match-value иЎЁзӨәжҹҘиҜўж—¶жҢүзү№е®ҡзҡ„еҖјиҝӣиЎҢжҹҘиҜўпјӣ

      • ж Үзӯҫ like иЎЁзӨәжҹҘиҜўж—¶жЁЎзіҠеҢ№й…Қпјӣ

      • ж Үзӯҫ like-left иЎЁзӨәе·ҰжЁЎзіҠеҢ№й…Қпјӣ

      • ж Үзӯҫ like-right иЎЁзӨәеҸіжЁЎзіҠеҢ№й…Қпјӣ

      • ж Үзӯҫ between иЎЁзӨәиҢғеӣҙжҹҘиҜўпјҢlow е’Ң high еҲҶеҲ«иЎЁзӨәжҹҘиҜўиҢғеӣҙзҡ„дёҠдёӢйҷҗпјӣиҫ№з•ҢеҖјжҳҜеҗҰеҢ…еҗ«еҸ–еҶідәҺж•°жҚ®еә“зұ»еһӢпјҲдҫӢеҰӮdb2жҳҜе·Ұй—ӯеҸіејҖпјүпјҢеҰӮжһңиҫ№з•ҢеҖјеҸӘдј дёҖдёӘ low жҲ–иҖ… highпјҢеҲҷжҢүеӨ§дәҺзӯүдәҺ low жҲ–иҖ…е°ҸдәҺзӯүдәҺ high жҹҘиҜўпјӣ

      • ж Үзӯҫ in иЎЁзӨәеҢ…еҗ«жҹҘиҜўпјҲжҹҘиҜўж•ҲзҺҮдҪҺпјҢж…Һз”ЁпјүпјҢе…¶еҸӮж•°еҖјдёҖдёӘж•°з»„пјҢеҰӮпјҡвҖқvalue1,value2,value3вҖқ.split(вҖң,вҖқ);

      • ж Үзӯҫ not-in иЎЁзӨәдёҚеҢ…еҗ«жҹҘиҜўпјҲжҹҘиҜўж•ҲзҺҮдҪҺпјҢж…Һз”ЁпјүпјҢе…¶еҸӮж•°еҖјдёҖдёӘж•°з»„пјҢеҰӮпјҡвҖқvalue1,value2,value3вҖқ.split(вҖң,вҖқ);

      • ж Үзӯҫ order-by иЎЁзӨәжҹҘиҜўз»“жһңжҢүиҜҘеӯ—ж®өиҝӣиЎҢжҺ’еәҸпјӣ

      • еұһжҖ§ column жҢҮзҡ„иЎЁзӨә Table зҡ„еӯ—ж®өеҗҚпјӣ

      • еұһжҖ§ javaType иЎЁзӨәж•°жҚ®еә“еӯ—ж®өзұ»еһӢеҜ№еә”еңЁ Java дёӯзҡ„ж•°жҚ®зұ»еһӢпјӣзјәзңҒиҜҘеұһжҖ§еҲҷй»ҳи®Өдёә Stringпјӣ

      • еұһжҖ§ searchType иЎЁзӨәиҜҘеӯ—ж®өеңЁ Controller жҺҘж”¶йЎөйқўдј жқҘзҡ„еҖјзҡ„зұ»еһӢпјӣзјәзңҒиҜҘеұһжҖ§еҲҷй»ҳи®Өдёә Stringпјӣ

      • еұһжҖ§ param иЎЁзӨәиҜҘеӯ—ж®өеңЁ Java д»Јз Ғдёӯзҡ„е‘ҪеҗҚпјҢд»Ҙй©јеі°еҪўејҸе‘ҪеҗҚпјӣзјәзңҒиҜҘеұһжҖ§еҲҷй»ҳи®Өдёә column зҡ„еҖјпјӣ

      • еұһжҖ§ asc зҡ„еҖјеҶіе®ҡдәҶжҹҘиҜўз»“жһңзҡ„жҺ’еәҸж–№ејҸпјҢfalse иЎЁзӨәжҢүйҷҚеәҸжҺ’еҲ—пјҢзјәзңҒиҜҘеұһжҖ§жҲ–иҖ…и®ҫзҪ®дёә true иЎЁзӨәжҢүеҚҮеәҸжҺ’еҲ—з»“жһңпјӣ
